# BACKEND

from flask import Flask, request, jsonify
import requests
import pandas as pd
from io import StringIO

app = Flask(__name__)

@app.route('/ema', methods=['GET'])
def get_ema():
    aandeelticker = request.args.get('aandeelticker')
    response = requests.get(f'https://eodhistoricaldata.com/api/eod/{aandeelticker}.US?api_token=jouw_api_sleutel&fmt=csv')
    data = pd.read_csv(StringIO(response.text))
    for span in [10, 20, 30]:
        data[f'ema{span}'] = data['Close'].ewm(span=span, adjust=False).mean()
    return data.to_json(orient='records')


# FRONTEND

import React, { useState, useEffect } from 'react';
import axios from 'axios';

function Ema() {
  const [aandeelticker, setAandeelticker] = useState('');
  const [ema, setEma] = useState([]);

  const haalEmaOp = async () => {
    const result = await axios.get(`/ema?aandeelticker=${aandeelticker}`);
    setEma(result.data);
  };

  useEffect(() => {
    haalEmaOp();
  }, [aandeelticker]);

  return (
    <div>
      <input type="text" value={aandeelticker} onChange={e => setAandeelticker(e.target.value)} placeholder="Aandeelticker" />
      {ema.map((dag, index) => (
        <p key={index}>Datum: {dag.Date}, Sluitingsprijs: {dag.Close}, EMA10: {dag.ema10}, EMA20: {dag.ema20}, EMA30: {dag.ema30}</p>
      ))}
    </div>
  );
}

export default Ema;
