from flask import Flask, request, jsonify
import requests

app = Flask(__name__)

@app.route('/historical_price', methods=['GET'])
def get_historical_price():
    aandeelticker = request.args.get('aandeelticker')
    response = requests.get(f'http://api.marketstack.com/v1/eod?access_key=jouw_api_sleutel&symbols={aandeelticker}')
    data = response.json()
    return jsonify(data)
