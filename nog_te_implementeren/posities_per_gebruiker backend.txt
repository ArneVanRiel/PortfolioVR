@app.route('/posities', methods=['GET'])
def haal_posities_op():
    if 'gebruiker_id' not in session:
        return jsonify({'message': 'Niet aangemeld'}), 401
    result = db.session.execute("""
        SELECT aandeelticker, SUM(eenheden) as totale_eenheden
        FROM Transacties
        WHERE gebruiker_id = :gebruiker_id
        GROUP BY aandeelticker
    """, {'gebruiker_id': session['gebruiker_id']})
    posities = [{'aandeelticker': row[0], 'totale_eenheden': row[1]} for row in result]
    return jsonify({'posities': posities}), 200

@app.route('/positie/<aandeelticker>', methods=['GET'])
def haal_positie_op(aandeelticker):
    if 'gebruiker_id' not in session:
        return jsonify({'message': 'Niet aangemeld'}), 401
    gegeven_datum = request.args.get('datum')
    result = db.session.execute("""
        SELECT aandeelticker, SUM(eenheden) as totale_eenheden
        FROM Transacties
        WHERE gebruiker_id = :gebruiker_id AND tijdstip <= :gegeven_datum AND aandeelticker = :aandeelticker
        GROUP BY aandeelticker
    """, {'gebruiker_id': session['gebruiker_id'], 'gegeven_datum': gegeven_datum, 'aandeelticker': aandeelticker})
    posities = [{'aandeelticker': row[0], 'totale_eenheden': row[1]} for row in result]
    return jsonify({'posities': posities}), 200

@app.route('/totale_positie', methods=['GET'])
def haal_totale_positie_op():
    if 'gebruiker_id' not in session:
        return jsonify({'message': 'Niet aangemeld'}), 401
    gegeven_datum = request.args.get('datum')
    result = db.session.execute("""
        SELECT SUM(eenheden) as totale_eenheden
        FROM Transacties
        WHERE gebruiker_id = :gebruiker_id AND tijdstip <= :gegeven_datum
    """, {'gebruiker_id': session['gebruiker_id'], 'gegeven_datum': gegeven_datum})
    totale_positie = result.scalar()
    return jsonify({'totale_positie': totale_positie}), 200
