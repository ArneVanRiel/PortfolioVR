import React, { useState } from 'react';
import axios from 'axios';

function Login() {
  const [gebruikersnaam, setGebruikersnaam] = useState('');
  const [wachtwoord, setWachtwoord] = useState('');

  const meldAan = async () => {
    const result = await axios.post('/login', { gebruikersnaam, wachtwoord });
    if (result.status === 200) {
      alert('Succesvol aangemeld');
    } else {
      alert('Onjuiste gebruikersnaam of wachtwoord');
    }
  };

  return (
    <div>
      <input type="text" value={gebruikersnaam} onChange={e => setGebruikersnaam(e.target.value)} placeholder="Gebruikersnaam of e-mail" />
      <input type="password" value={wachtwoord} onChange={e => setWachtwoord(e.target.value)} placeholder="Wachtwoord" />
      <button onClick={meldAan}>Meld aan</button>
    </div>
  );
}

export default Login;
