import React, { useState, useEffect } from 'react';
import axios from 'axios';
import { Line } from 'react-chartjs-2';

function PortfolioWaardeGrafiek() {
  const [aandeelticker, setAandeelticker] = useState('');
  const [data, setData] = useState({});

  const haalPortfolioWaardeOp = async () => {
    const result = await axios.get(`/portfolio_waarde?aandeelticker=${aandeelticker}`);
    const waardes = result.data;
    const labels = waardes.map(waarde => waarde.datum);
    const dataset = waardes.map(waarde => waarde.waarde);
    setData({
      labels: labels,
      datasets: [
        {
          label: 'Portfolio Waarde',
          data: dataset,
          borderColor: 'rgba(75,192,192,1)',
          fill: false
        }
      ]
    });
  };

  useEffect(() => {
    haalPortfolioWaardeOp();
  }, [aandeelticker]);

  return <Line data={data} />;
}

export default PortfolioWaardeGrafiek;
