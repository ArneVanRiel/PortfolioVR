SELECT aandeelticker, SUM(eenheden) as totale_eenheden
FROM Transacties
WHERE gebruiker_id = :gebruiker_id
GROUP BY aandeelticker;


-- Positie van een bepaalde aandeelticker per gebruiker
SELECT aandeelticker, SUM(eenheden) as totale_eenheden
FROM Transacties
WHERE gebruiker_id = :gebruiker_id AND tijdstip <= :gegeven_datum AND aandeelticker = :aandeelticker
GROUP BY aandeelticker;

-- Totale positie in alle aandelen
SELECT SUM(eenheden) as totale_eenheden
FROM Transacties
WHERE gebruiker_id = :gebruiker_id AND tijdstip <= :gegeven_datum;
