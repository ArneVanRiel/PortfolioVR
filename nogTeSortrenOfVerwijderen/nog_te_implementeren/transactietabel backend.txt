from flask import Flask, request, jsonify, session
from flask_sqlalchemy import SQLAlchemy
from werkzeug.security import check_password_hash
from datetime import datetime

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:////tmp/test.db'
app.config['SECRET_KEY'] = 'jouw geheime sleutel'
db = SQLAlchemy(app)

class Gebruiker(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    gebruikersnaam = db.Column(db.String(80), unique=True, nullable=False)
    wachtwoord = db.Column(db.String(120), nullable=False)

class Transactie(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    gebruiker_id = db.Column(db.Integer, nullable=False)
    aandeelticker = db.Column(db.String(120), nullable=False)
    eenheden = db.Column(db.Float, nullable=False)
    hoeveelheid = db.Column(db.Float, nullable=False)
    tijdstip = db.Column(db.DateTime, nullable=False)

@app.route('/login', methods=['POST'])
def login():
    data = request.get_json()
    gebruiker = Gebruiker.query.filter_by(gebruikersnaam=data['gebruikersnaam']).first()
    if gebruiker and check_password_hash(gebruiker.wachtwoord, data['wachtwoord']):
        session['gebruiker_id'] = gebruiker.id
        return jsonify({'message': 'Succesvol aangemeld'}), 200
    else:
        return jsonify({'message': 'Onjuiste gebruikersnaam of wachtwoord'}), 401

@app.route('/transacties', methods=['POST'])
def voeg_transactie_toe():
    if 'gebruiker_id' not in session:
        return jsonify({'message': 'Niet aangemeld'}), 401
    data = request.get_json()
    nieuwe_transactie = Transactie(
        gebruiker_id=session['gebruiker_id'],
        aandeelticker=data['aandeelticker'],
        eenheden=data['eenheden'],
        hoeveelheid=data['hoeveelheid'],
        tijdstip=datetime.strptime(data['tijdstip'], '%Y-%m-%d %H:%M:%S')
    )
    db.session.add(nieuwe_transactie)
    db.session.commit()
    return jsonify({'message': 'Transactie toegevoegd'}), 201
